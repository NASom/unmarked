crPiFun <- function(p) { # p should have 3 columns
    cbind("001" = (1-p[,1]) * (1-p[,2]) * p[,3],
          "010" = (1-p[,1]) * p[,2]     * (1-p[,3]),
          "011" = (1-p[,1]) * p[,2]     * p[,3],
          "100" = p[,1]     * (1-p[,2]) * (1-p[,3]),
          "101" = p[,1]     * (1-p[,2]) * p[,3],
          "110" = p[,1]     * p[,2]     * (1-p[,3]),
          "111" = p[,1]     * p[,2]     * p[,3])
}
